pub fn balanced_string_split(s: String) -> i32 {
  let mut count = 0;
  let mut count_l = 0;
  let mut count_r = 0;
  for c in s.chars() {
    if c == 'L' {
      count_l += 1;
    } else {
      count_r += 1;
    }
    if count_l == count_r {
      count += 1
    }
  }
  count
}
