# 字符串转换整数 (atoi)

[leetcode](https://leetcode-cn.com/problems/string-to-integer-atoi/)

## Solution

### 纯调试

```js
/**
 * @param {string} s
 * @return {number}
 */
var myAtoi = function (s) {
  const range = [-2147483648, 2147483647]
  let phase = ''
  const startSet = '+-0123456789'
  const numSet = '0123456789'
  let ret = ''
  for (let i = 0; i < s.length; i++) {
    const v = s[i];
    if (!phase) {
      if (v === ' ') { continue }
      if (startSet.indexOf(v) > -1) {
        if (v === '0') {
          phase = '0'
        } else {
          if (v !== '+') { ret += v }
          phase = 'started'
        }
        continue
      } else {
        return 0
      }
    }
    if (phase === '0') {
      if (v === '0') {
        continue
      } else if (numSet.indexOf(v) > -1) {
        phase = 'started'
        ret += v
        continue
      } else {
        return 0
      }
    }
    if (phase === 'started') {
      if (numSet.indexOf(v) > -1) {
        ret += v
      } else {
        break
      }
    }
  }


  // remove useless 0
  ret = ret.replace(/^(-)?(0+)(\d*)/, '$1$3') || '0'

  // defend '+' '-'
  if (ret.length === 1 && '+-'.indexOf(ret[0]) > -1) {
    return 0
  }


  // suppose Number(ret) directly will overflow
  // check range
  if (ret[0] === '-') {
    if (ret.length - 1 > 10) { return range[0] }
    if (ret.length - 1 === 10) {
      const min = range[0] + ''
      for (let i = 0; i < ret.length; i++) {
        const v = ret[i];
        const m = min[i]
        if (v > m) { return range[0] }
        if (v === m) { continue }
        if (v < m) { break }
      }
    }
  } else {
    if (ret.length > 10) { return range[1] }
    if (ret.length === 10) {
      const max = range[1] + ''
      for (let i = 0; i < ret.length; i++) {
        const v = ret[i];
        const m = max[i]
        if (v > m) { return range[1] }
        if (v === m) { continue }
        if (v < m) { break }
      }
    }
  }
  return Number(ret)
};

```
